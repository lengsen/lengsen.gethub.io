异或软件加密程序

实验目的

使用异或的方式对文件进行加密

各模块的描述

目主体模块：即main()函数，主要的功能为：确认密码与展示主界面

密码的确认模块：即checkPwd()函数，作用是确认用户输入的密码是否正确，正确返回1，错误返回0，检测到正确密码后程序才会执行下一步功能

展示主界面模块：即showFunction()函数，作用是展示加解密，更改密码，退出系统等功能供用户选择

加解密模块：即encrypt()函数和encode()函数，实际上都调用了异或加解密模块，相当于是对后者的二次利用

异或加解密模块：即encodeOrDecodeFile()函数，该函数先是分批原文件，再对分批的数据流使用异或加解密，即异或一次等于加密，异或两次等与解密，最后写入到目标文件中

更改密码模块：即changeCode()函数，用以更新用户的密码，采用双重校验的手段确保用户的新密码不存在手误输入

// 校验密码
// 返回 0 表示校验失败，返回 1 表示校验成功
int checkPwd(){
    int pwd;  //用户输入的密码
    int trytimes;  // 用户尝试输入密码的次数

    printf("程序设置了密码，验证通过后才能使用。请输入6位数字密码：");
    // 可以输入三次
    for(trytimes=1; trytimes<=3; trytimes++){
        scanf("%d",&pwd);
        fflush(stdin);
        if(pwd==password){
            return 1;
            break;
        }else if(trytimes==3){
            return 0;
        }else{
            printf("抱歉，密码错误，您还有%d次机会：", 3-trytimes);
        }
    }
}

void encrypt(){
    char sourcefile[30],  // 加密的文件名
            targetFile[30],  // 加密后要保存的文件名
            secretKey[21];  // 文件加密的密钥
    printf("输入要加密的文件名(含路径)：");
    scanf("%s", sourcefile);
    printf("输入密钥：");  //密钥是用户自己定义的，可以随意给需要加密的文件添加密钥
    scanf("%s", secretKey);
    printf("加密后的文件名(含路径)：");  //给加密后的文件命名，并保存
    scanf("%s",targetFile);
    if( encrORdocoFile(sourcefile, secretKey, targetFile) ){
        printf("恭喜你，文件[%s]加密成功，保存在[%s]。\n", sourcefile, targetFile);
    }
    else
    printf("加密失败\n"); 
}
void decode(){
    char sourcefile[50],  // 解密的文件名
            targetFile[50],  // 加密后要保存的文件名
            secretKey[21];  // 文件解密的密钥
    printf("输入要解密的文件名(含路径)：");
    scanf("%s", sourcefile);
    printf("输入密钥：");  //密钥是用户自己定义的，可以随意给需要加密的文件添加密钥
    scanf("%s", secretKey);
    printf("解密后的文件名(含路径)：");  //给加密后的文件命名，并保存
    scanf("%s",targetFile);
    if( encrORdocoFile(sourcefile, secretKey, targetFile) ){
        printf("恭喜你，文件[%s]解密成功，保存在[%s]。\n", sourcefile, targetFile);
    }
    else
    printf("解密失败\n"); 
}
//加密or解密 
int encrORdocoFile(char *sourcefile, char *secretKey, char *targetFile){
    FILE *fpSource, *fpTarget;  // 要打开的文件的指针
    char buffer[21];  // 缓冲区，用于存放从文件读取的数据
    int readCount,  // 每次从文件中读取的字节数
        keyLen = strlen(secretKey),  // 密钥的长度
        i;  // 循环次数
    // 以二进制方式读取/写入文件
    fpSource = fopen(sourcefile, "r");
    if(fpSource==NULL){
        printf("文件[%s]打开失败，请检查文件路径和名称是否输入正确！\n", sourcefile);
        return 0;
    }
    fpTarget = fopen(targetFile, "w");
    if(fpTarget==NULL){
        printf("文件[%s]创建/写入失败！请检查文件路径和名称是否输入正确！\n", fpTarget);
        return 0;
    }
    // 不断地从文件中读取 keyLen 长度的数据，保存到buffer，直到文件结束
    while( (readCount=fread(buffer, 1, keyLen, fpSource)) > 0 ){
        // 对buffer中的数据逐字节进行异或运算
        for(i=0; i<readCount; i++){
            buffer[i] ^= secretKey[i];
        }
        // 将buffer中的数据写入文件
        fwrite(buffer, 1, readCount, fpTarget);
    }
    fclose(fpSource);
    fclose(fpTarget);
    return 1;
}
int changeCode(){
    int pwd;  //用户输入的old密码
    int pwdt;    //用户的new密码 
    int pwds;    //用户的new2密码 
    int trytimes;  // 用户尝试输入密码的次数
    printf("重置密码前必须验证原密码。请输入6位数字原密码：");
    for(trytimes=1; trytimes<=3; trytimes++){
        scanf("%d",&pwd);
        fflush(stdin);
        if(pwd==password){
            printf("请输入新密码：");
            scanf("%d",&pwdt);
            printf("请再次输入新密码：");
            scanf("%d",&pwds);
            if(pwds==pwdt){
                password=pwd;
                printf("恭喜你，密码修改成功！\n");
                return 1;
                break;
            }
            else{
                return 0; 
            }
        }else if(trytimes==3){
            return 0;
        }else{
            printf("抱歉，密码错误，您还有%d次机会：", 3-trytimes);
        }
    }
}
